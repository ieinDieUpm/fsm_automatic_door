40 void port_led_timer_setup(port_led_hw_t *p_led)
41 {
42  // Enable the peripheral clock
43  if (p_led->p_timer == TIM3)
44  {
45  RCC->APB1ENR |= RCC_APB1ENR_TIM3EN;
46  }
47  else if (p_led->p_timer == TIM4)
48  {
49  RCC->APB1ENR |= RCC_APB1ENR_TIM4EN;
50  }
51 
52  // Disable the timer
53  p_led->p_timer->CR1 &= ~TIM_CR1_CEN;
54 
55  // Autoreload preload enabled
56  p_led->p_timer->CR1 |= TIM_CR1_ARPE;
57 
58  // Reset the values of the timer
59  p_led->p_timer->CNT = 0;
60 
61  // Set the timeout value
62  // Compute ARR and PSC to match the duration in milliseconds. Check if the duration is too long and adapt prescaler and ARR
63  double sec = (double)p_led->timer_blink_semi_period_ms / 1000.0;
64  double scc = (double)SystemCoreClock;
65  double psc = round(((scc * sec) / (65535.0 + 1.0)) - 1.0);
66  double arr = round(((scc * sec) / (psc + 1.0)) - 1.0);
67 
68  // Adjust psc and arr if necessary
69  if (arr > 0xFFFF)
70  {
71  psc += 1.0;
72  arr = round((scc * sec) / (psc + 1.0) - 1.0);
73  }
74 
75  // Load the values
76  p_led->p_timer->ARR = (uint32_t)(round(arr));
77  p_led->p_timer->PSC = (uint32_t)(round(psc));
78 
79  // Clean interrupt flags
80  p_led->p_timer->SR &= ~TIM_SR_UIF;
81 
82  // Enable the update interrupt
83  p_led->p_timer->DIER |= TIM_DIER_UIE;
84 
85  if (p_led->p_timer == TIM3)
86  {
87  // Enable the interrupt in the NVIC
88  NVIC_SetPriority(TIM3_IRQn, NVIC_EncodePriority(NVIC_GetPriorityGrouping(), 2, 0)); /* Priority 2, sub-priority 0 */
89  NVIC_EnableIRQ(TIM3_IRQn);
90  }
91  else if (p_led->p_timer == TIM4)
92  {
93  // Enable the interrupt in the NVIC
94  NVIC_SetPriority(TIM4_IRQn, NVIC_EncodePriority(NVIC_GetPriorityGrouping(), 2, 0)); /* Priority 2, sub-priority 0 */
